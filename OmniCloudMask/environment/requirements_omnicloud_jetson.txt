#   conda create -n omnicloud python=3.10 -y
#   conda activate omnicloud
#   pip install torch-2.5.0-cp310-cp310-linux_aarch64.whl   (NVIDIA aarch64 wheel)
#   pip install omnicloudmask==1.7.1
#   pip install segmentation-models-pytorch timm
#   export PYTHONPATH=~/timm_stub:$PYTHONPATH   (reuse the timm stub from CD-Mamba)
#
# Weights are downloaded automatically from HuggingFace Hub on first call.
